① Create Google Cloud Project
1Go to console.cloud.google.com → sign in with the Google account that owns your YouTube channel
2Top bar → Select a project → New Project → name it yt-automation → Create
3Left menu → APIs & Services → Library → search YouTube Data API v3 → click it → Enable
② OAuth Consent Screen
4APIs & Services → OAuth consent screen → select External → Create
5Fill in App name, User support email, Developer email → Save and Continue through all steps
6Test users → + Add Users → add your Gmail → Save. Required while in Testing mode.
③ Pipeline — Desktop App credentials
7Credentials → + Create Credentials → OAuth 2.0 Client ID
8Type = Desktop app → name "Pipeline" → Create
9Download JSON → rename to client_secrets.json → place in project root
10Run python main.py → browser opens → Allow → token saved
④ Dashboard — Web App credentials
11Credentials → + Create Credentials → OAuth 2.0 Client ID again
12Type = Web application → name "Dashboard"
13Authorized JavaScript origins → add http://localhost:8080 → Create
14Copy the Client ID → paste into the field above → Save Settings
15Click Connect YouTube → browser popup → Allow → status turns green
⚠ Common issues:
• "Access blocked" — add your email as a Test User (step 6)
• "redirect_uri_mismatch" — add http://localhost:8080 to Authorized JavaScript origins
• Token expires 7 days (Testing mode) — publish the app or re-authorize
• Daily quota = 10,000 units free (6 uploads/day)
• Need more? Request quota increase (free, 1-2 days)

import schedule,time,json,os from dotenv import load_dotenv import anthropic from main import run load_dotenv() NICHE=os.getenv("CHANNEL_NICHE","politics and religion") AUDIENCE=os.getenv("CHANNEL_AUDIENCE","global") LANG=os.getenv("CHANNEL_LANGUAGE","english") FREQ=int(os.getenv("VIDEOS_PER_WEEK",5)) QUEUE_F="topic_queue.json" client=anthropic.Anthropic(api_key=os.getenv("ANTHROPIC_API_KEY")) def load_q(): return json.load(open(QUEUE_F)) if os.path.exists(QUEUE_F) else [] def save_q(q):json.dump(q,open(QUEUE_F,"w"),indent=2) def refill(): q=load_q() if len(q)>=5:return print("Refilling queue...") msg=client.messages.create(model="claude-sonnet-4-20250514",max_tokens=1000, messages=[{"role":"user","content":f""" Generate 10 trending YouTube topics for {NICHE} channel. Audience:{AUDIENCE}. Language:{LANG}. Return ONLY a JSON array of strings. Make them clickable and searchable."""}]) raw=msg.content[0].text new=json.loads(raw[raw.index("["):raw.rindex("]")+1]) q.extend(new) save_q(q) print(f"Added {len(new)} topics") def publish_next(): refill() q=load_q() if not q:return topic=q.pop(0) save_q(q) try:run(topic) except Exception as e: print(f"Error: {e}") q.insert(0,topic);save_q(q) # 14:00 UTC = global peak engagement window (US morning, EU afternoon) if FREQ>=7:schedule.every().day.at("14:00").do(publish_next) elif FREQ>=5: for d in["monday","tuesday","wednesday","thursday","friday"]: getattr(schedule.every(),d).at("14:00").do(publish_next) else: for d in["monday","wednesday","friday"]: getattr(schedule.every(),d).at("14:00").do(publish_next) print(f"Scheduler running - {FREQ} videos/week") refill() while True: schedule.run_pending() time.sleep(30)

""" SQLite database layer for YouTube automation. Tracks: video history, API costs, performance metrics. Zero config — creates yt_automation.db automatically. """ import sqlite3, json, os from datetime import datetime, date from pathlib import Path DB_PATH = os.getenv("DB_PATH", "yt_automation.db") def get_conn(): conn = sqlite3.connect(DB_PATH) conn.row_factory = sqlite3.Row return conn def init_db(): """Create tables if they don't exist. Safe to call on every startup.""" conn = get_conn() conn.executescript(""" CREATE TABLE IF NOT EXISTS videos ( id INTEGER PRIMARY KEY AUTOINCREMENT, topic TEXT NOT NULL, title TEXT, youtube_id TEXT UNIQUE, status TEXT DEFAULT 'pending', niche TEXT, language TEXT, duration_s INTEGER, created_at TEXT DEFAULT (datetime('now')), published_at TEXT, thumbnail TEXT ); CREATE TABLE IF NOT EXISTS performance ( id INTEGER PRIMARY KEY AUTOINCREMENT, video_id INTEGER REFERENCES videos(id), recorded_at TEXT DEFAULT (datetime('now')), views INTEGER DEFAULT 0, watch_mins INTEGER DEFAULT 0, likes INTEGER DEFAULT 0, comments INTEGER DEFAULT 0, subs_gained INTEGER DEFAULT 0, cpm REAL DEFAULT 0, revenue REAL DEFAULT 0 ); CREATE TABLE IF NOT EXISTS api_costs ( id INTEGER PRIMARY KEY AUTOINCREMENT, service TEXT NOT NULL, operation TEXT, units REAL DEFAULT 0, cost_usd REAL DEFAULT 0, video_id INTEGER REFERENCES videos(id), recorded_at TEXT DEFAULT (datetime('now')) ); CREATE TABLE IF NOT EXISTS topic_queue ( id INTEGER PRIMARY KEY AUTOINCREMENT, topic TEXT NOT NULL, type TEXT DEFAULT 'Planned', scheduled TEXT, status TEXT DEFAULT 'pending', added_at TEXT DEFAULT (datetime('now')) ); """) conn.commit() conn.close() print(f"DB ready: {DB_PATH}") # ── Videos ──────────────────────────────────────────────────── def log_video_start(topic, niche, language): conn = get_conn() cur = conn.execute( "INSERT INTO videos (topic, niche, language, status) VALUES (?,?,?,?)", (topic, niche, language, 'generating') ) vid_id = cur.lastrowid conn.commit(); conn.close() return vid_id def log_video_complete(vid_id, title, youtube_id, duration_s): conn = get_conn() conn.execute( "UPDATE videos SET title=?, youtube_id=?, status='published', " "duration_s=?, published_at=datetime('now') WHERE id=?", (title, youtube_id, duration_s, vid_id) ) conn.commit(); conn.close() def log_video_error(vid_id, error_msg): conn = get_conn() conn.execute( "UPDATE videos SET status='error', title=? WHERE id=?", (f"ERROR: {error_msg[:200]}", vid_id) ) conn.commit(); conn.close() def get_video_history(limit=50): conn = get_conn() rows = conn.execute( "SELECT * FROM videos ORDER BY created_at DESC LIMIT ?", (limit,) ).fetchall() conn.close() return [dict(r) for r in rows] # ── Performance ─────────────────────────────────────────────── def log_performance(video_id, views, watch_mins, likes=0, comments=0, subs_gained=0, cpm=1.5): revenue = (views / 1000) * cpm conn = get_conn() conn.execute( "INSERT INTO performance (video_id,views,watch_mins,likes," "comments,subs_gained,cpm,revenue) VALUES (?,?,?,?,?,?,?,?)", (video_id, views, watch_mins, likes, comments, subs_gained, cpm, revenue) ) conn.commit(); conn.close() return revenue def get_channel_stats(): conn = get_conn() stats = conn.execute(""" SELECT COUNT(DISTINCT v.id) AS total_videos, COALESCE(SUM(p.views),0) AS total_views, COALESCE(SUM(p.watch_mins)/60.0,0) AS total_watch_hours, COALESCE(SUM(p.subs_gained),0) AS total_subs, COALESCE(SUM(p.revenue),0) AS total_revenue, COALESCE(AVG(p.views),0) AS avg_views FROM videos v LEFT JOIN performance p ON p.video_id = v.id WHERE v.status = 'published' """).fetchone() conn.close() return dict(stats) # ── API Costs ───────────────────────────────────────────────── def log_cost(service, operation, units, cost_usd, video_id=None): conn = get_conn() conn.execute( "INSERT INTO api_costs (service,operation,units,cost_usd,video_id) " "VALUES (?,?,?,?,?)", (service, operation, units, cost_usd, video_id) ) conn.commit(); conn.close() def get_monthly_costs(year=None, month=None): now = datetime.now() y = year or now.year m = month or now.month conn = get_conn() rows = conn.execute(""" SELECT service, SUM(cost_usd) AS total_cost, SUM(units) AS total_units, COUNT(*) AS calls FROM api_costs WHERE strftime('%Y-%m', recorded_at) = ? GROUP BY service ORDER BY total_cost DESC """, (f"{y:04d}-{m:02d}",)).fetchall() conn.close() return [dict(r) for r in rows] def get_total_spent(): conn = get_conn() row = conn.execute( "SELECT COALESCE(SUM(cost_usd),0) AS total FROM api_costs" ).fetchone() conn.close() return row['total'] # ── YPP progress ────────────────────────────────────────────── def get_ypp_progress(): stats = get_channel_stats() subs = stats['total_subs'] hours = stats['total_watch_hours'] return { 'subs': subs, 'subs_target': 1000, 'subs_pct': min(100, round(subs/10)), 'watch_hours': round(hours, 1), 'watch_target': 4000, 'watch_pct': min(100, round(hours/40)), 'eligible': subs >= 1000 and hours >= 4000 } # ── Run on import ───────────────────────────────────────────── init_db() # ── Usage: integrate into main.py ──────────────────────────── # from database import log_video_start, log_video_complete, log_cost # # def run(topic): # vid_id = log_video_start(topic, NICHE, LANGUAGE) # content = generate_content(topic) # log_cost('anthropic', 'script', 3000, 0.07, vid_id) # audio_files = generate_audio(content) # log_cost('elevenlabs', 'tts', 4500, 0.05, vid_id) # image_files = fetch_visuals(content) # log_cost('pexels', 'search', 12, 0.0, vid_id) # thumb = make_thumbnail(content, image_files[0]) # video = build_video(content, audio_files, image_files) # yt_id = upload(video, thumb, content) # log_video_complete(vid_id, content['title'], yt_id, 360) # return yt_id

Security — how your keys are stored
Personal use safe
💻 This browser tool
API keys are read from your .env file at startup and stored server-side in SQLite — never in the browser. Keys are only transmitted directly to each API endpoint (anthropic.com, elevenlabs.io, etc.) when you click a button.
📄 Python .env file
Keys live in a plain text .env file on your computer. Python reads them at startup. Never commit .env to GitHub. Add it to .gitignore immediately. Use the Export button below to generate it.
⚠ Treat API keys like passwords. Rotate them immediately if you ever share your screen or upload code publicly.
